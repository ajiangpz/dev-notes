# 描述
找出两个数组的最长重复子数组

# 思路

动态规划 构建 DP表 ，条件是相等时加1

```js
var findLength = function(nums1, nums2) {
    const m = nums1.length, n = nums2.length;
    const dp = Array(m + 1).fill(0).map(() => Array(n + 1).fill(0));
    let maxLength = 0;

    for (let i = 1; i <= m; i++) {
        for (let j = 1; j <= n; j++) {
            if (nums1[i - 1] === nums2[j - 1]) {
                dp[i][j] = dp[i - 1][j - 1] + 1;
                maxLength = Math.max(maxLength, dp[i][j]);
            }
        }
    }

    return maxLength;
};
```
