# 描述

从 left 到 right 反转链表

# 实现

```js
var reverseBetween = function (head, left, right) {
  if (!head || left === right) return head;

  let dummy = new ListNode(0);
  dummy.next = head;
  let pre = dummy;
  // 找到开始节点的前一个节点
  for (let i = 1; i < left; i++) {
    pre = pre.next;
  }
  // 开始节点
  let cur = pre.next;
  let prev = null;
  for (let i = 0; i <= right - left; i++) {
    let next = cur.next;
    cur.next = prev;
    prev = cur;
    cur = next;
  }
  // 反转后结束节点指向下个节点
  pre.next.next = cur;
  pre.next = prev;

  return dummy.next;
};
```
